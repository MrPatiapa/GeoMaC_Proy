Convección-difusión de calor estacionaria¶

Considere el siguiente problema:

$$ \begin{eqnarray*} c_p \rho \frac{\partial}{\partial x} \left( u T \right) - \frac{\partial }{\partial x} \left( \kappa \frac{\partial T}{\partial x}\right) & = & S \\ T(0) & = & 1 \\ T(L) & = & 0 \end{eqnarray*} $$

La solucin analtica es la siguiente:

$$ \displaystyle T(x) = \frac{\exp\left(\frac{\rho u x}{\kappa}\right) - 1 }{\exp\left(\frac{\rho v L}{\kappa}\right) - 1} (T_L - T_0) + T_0 $$

Implementar la solucin numrica con diferencias finitas en Python.

Utilice los siguientes datos:

  • $L = 1.0$ [m],
  • $c_p = 1.0$ [J / Kg $^\text{o}$K],
  • $\rho = 1.0$ [kg/m$^3$],
  • $\kappa = 0.1$ [kg/m s],
  • $S = 0$

def mesh(L,N):
    """
    Esta funcin calcula el h y las coordenadas de la malla
    
    Parameters
    ----------
    L : float
    Longitud del dominio.
    
    N : int
    Nmero de incgnitas (sin las fronteras)
    
    Returns
    -------
    h, x: el tamao h de la malla y las coordenadas en la direccin x
    """
    # YOUR CODE HERE
    h = L / (N+1)  # tamao h de la malla
    x = np. linspace (0, L, N+2) # Coordenadas X
    
    return h, x

def Laplaciano1D(par):
    """
    Esta funcin calcula los coeficientes de la matriz de 
    diferencias finitas.
    
    Paremeters
    ----------
    par: dict
    Diccionario que contiene todos los datos del problema.
    
    Returns
    ----------
    A : la matriz de la discretizacin.
    """
    N     = par['N'] 
    h     = par['h']
    alpha = par['alpha']
    cp    = par['cp']
    rho   = par['rho']
    u     = par['u'] 

    # YOUR CODE HERE
    b = -(u / (2*h)) + (alpha / (h*h)) 
    c = (u / (2*h)) + (alpha / (h*h))
        
    a = b + c
    A = np.zeros((N,N))
    A[0,0] = a 
    A[0,1] = -b
    for i in range(1,N-1):
        A[i,i] = a 
        A[i,i+1] = -b 
        A[i,i-1] = -c 
    A[N-1,N-2] = -c
    A[N-1,N-1] = a
    
    return A

def RHS(par):
    """
    Esta funcin calcula el lado derecho del sistema lineal.
    
    Paremeters
    ----------
    par: dict
    Diccionario que contiene todos los datos del problema.
    
    Returns
    ----------
    f : el vector del lado derecho del sistema.
    """
    N     = par['N'] 
    h     = par['h']
    alpha = par['alpha']
    cp    = par['cp']
    rho   = par['rho']
    u     = par['u'] 
    T0    = par['BC'][0]
    TL    = par['BC'][1]
    
    f = np.zeros(N)       

    # YOUR CODE HERE
    S = 0
    b = -(u / (2*h)) + (alpha / (h*h)) 
    c = (u / (2*h)) + (alpha / (h*h))
    
    f[:] = S/(cp*rho)
    
    f[0]  = (c * T0)
    f[N-1] = (b * TL)
    
    return f

def analyticSol(par, NP = 100):
    """
    Calcula la solucin analtica
    
    Paremeters
    ----------
    par: dict
    Diccionario que contiene todos los datos del problema.
    
    NP: int
    Nmero de puntos para calcular la solucin analtica. Si no se da
    ningn valor usa 100 puntos para hacer el clculo.
    
    Returns
    ----------
    xa, Ta : un arreglo (xa) con las coordenadas donde se calcula la 
       solucin analtica y otro arreglo (Ta) con la solucin analtica.
    """
    L     = par['L']
    rho   = par['rho']
    u     = par['u']
    alpha = par['alpha']
    T0    = par['BC'][0]
    TL    = par['BC'][1]

    # YOUR CODE HERE
    xa = np.linspace(0, L, NP) # Coordenadas X
    Ta = ((np.exp((rho*u*xa) / alpha)-1) * (TL-T0) /(np.exp((rho*u*L)/alpha)-1))+T0
    
    return xa, Ta

def numSol(par):
    """
    Funcin que calcula la matriz del sistema (A), el lado derecho (f)
    y con esta informacin resuelve el sistema lineal para obtener la 
    solucin.
    
    Paremeters
    ----------
    par: dict
    Diccionario que contiene todos los datos del problema.
    
    Returns
    ----------
    T : un arreglo (T) con la solucin analtica.
    """
    
    # YOUR CODE HERE
    N     = par['N']
    A = Laplaciano1D(par)
    f = RHS(par)
    
    T= np.zeros(N)  # Vector de almacenamiento
    T= np.linalg.solve (A,f)
    
    return T

def error(Ta, Tn):
    """
    Funcin que calcula el error de la solucin numrica.
    
    Paremeters
    ----------
    Ta: array
    Arreglo con la solucin analtica.
    
    T: array
    Arreglo con la solucin numrica.
    
    Returns
    ----------
    E : float
    Error de la solucin numrica con respecto a la analtica.
    """
    # YOUR CODE HERE
    E = np.linalg.norm(Ta-Tn,2)
    return E

def casos(u, N):
    """
    Funcin para resolver cada caso que usa las funciones anteriores.

    Paremeters
    ----------
    u: float
    Velocidad.
    
    N: int
    Nmero de incgnitas.
    """    
    # Definicin de un diccionario para almancenar los datos del problema
    par = {}
    par['L']     = 1.0 # m
    par['cp']    = 1.0 # [J / Kg K]
    par['rho']   = 1.0 # kg/m^3
    par['u']     = u   # m/s
    par['alpha'] = 0.1 # kg / m.s
    par['BC']    = (1.0, 0.0) # Condiciones de frontera
    par['N']     = N   # Nmero de incgnitas
    h, x = mesh(par['L'], par['N'])
    par['h'] = h

    # Definicin del arreglo donde se almacenar la solucin numrica
    N  = par['N']
    T        = np.zeros(N+2)
    T[0]     = par['BC'][0]  # Condicin de frontera en x = 0
    T[N+1]   = par['BC'][1]  # Condicin de frontera en x = L

    # Se ejecuta la funcin para obtener la solucin
    T[1:N+1] = numSol(par)

    # Se calcula la funcin para calcular la solucin analtica
    _, Ta = analyticSol(par, N+2)

    # Se calcula el error
    Error = error(Ta, T)

    # Se grafica la solucin
    plotSol(par, x, T, Error)
